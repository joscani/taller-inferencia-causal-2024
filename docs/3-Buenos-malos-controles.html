x <- rnorm(800)
y <- rnorm(800)

z <- as.factor(rbinom(800,1, plogis(2*x - 2*y + rnorm(800))))

d <- tibble(x, y, z)

d  |> 
 ggplot(aes(x = x, y = y)) +
  geom_point(aes(color = z)) +
  geom_smooth(method = "lm", se = FALSE, color = "black", linewidth= 2) +
  geom_smooth(aes(color= z), method = "lm", se = FALSE)
